cmake_minimum_required(VERSION 3.12)

#可选 包名不匹配警告
cmake_policy(SET CMP0114 OLD)

# vcpkg 工具链 — 通过预设或环境变量 VCPKG_ROOT 指定
if(NOT DEFINED CMAKE_TOOLCHAIN_FILE AND DEFINED ENV{VCPKG_ROOT})
    set(CMAKE_TOOLCHAIN_FILE "$ENV{VCPKG_ROOT}/scripts/buildsystems/vcpkg.cmake" CACHE STRING "")
endif()
if(NOT EXISTS "${CMAKE_TOOLCHAIN_FILE}")
    message(FATAL_ERROR "vcpkg toolchain not found: ${CMAKE_TOOLCHAIN_FILE}. Set VCPKG_ROOT environment variable.")
endif()

project(learnVTK LANGUAGES CXX)
set(CMAKE_CXX_STANDARD 20)

# 1. 设置输出目录
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY_DEBUG ${CMAKE_SOURCE_DIR}/app/bin)
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY_RELEASE ${CMAKE_SOURCE_DIR}/app/binr)
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)


# 设置CMake模块路径
list(APPEND CMAKE_MODULE_PATH ${CMAKE_SOURCE_DIR}/cmake)
# 2. 包含公用函数
include(DeployQt)


# 3. 查找依赖（用vcpkg自动查找，无需自定义Find模块）
find_package(VTK REQUIRED )
find_package(Qt6 REQUIRED )
find_package(CGNS REQUIRED )
